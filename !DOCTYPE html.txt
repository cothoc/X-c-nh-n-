<script>
    let current = 'cicilia';
    function switchHotel(h) {
        current = h;
        document.querySelectorAll('.hotel-form').forEach(f => f.classList.add('hidden'));
        document.getElementById('form-'+h).classList.remove('hidden');
        document.getElementById('out_hotel').innerText = h === 'cicilia' ? 'CICILIA HOTELS & SPA' : 'AQUAMARINE RESORT';
        update();
    }
    function update() {
        if(current === 'cicilia') {
            document.getElementById('out_ten').innerText = document.getElementById('c_ten').value;
            document.getElementById('out_in').innerText = document.getElementById('c_checkin').value;
            document.getElementById('out_out').innerText = document.getElementById('c_checkout').value;
            document.getElementById('out_loai').innerText = document.getElementById('c_loai').value;
        } else {
            document.getElementById('out_ten').innerText = document.getElementById('a_ten').value;
            document.getElementById('out_in').innerText = document.getElementById('a_checkin').value;
            document.getElementById('out_out').innerText = document.getElementById('a_checkout').value;
            document.getElementById('out_loai').innerText = 'Combo: ' + document.getElementById('a_combo').value;
        }
    }
    function copyImage() {
        html2canvas(document.querySelector("#phieu")).then(canvas => {
            canvas.toBlob(blob => {
                navigator.clipboard.write([new ClipboardItem({"image/png": blob})]);
                alert("Đã copy ảnh! Dán vào Zalo ngay.");
            });
        });
    }
</script>
